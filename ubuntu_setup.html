Ubuntu Setup.

This is a briefly annotated account of how we set up a Ruby & Rails Environment on a fresh install of Ubuntu 10.04 LTS (Lucid Lynx).

# retrieve the latest packages
$ sudo apt-get update

# update the current packages
$ sudo apt-get upgrade

# allow configuration of newly updated packages
$ restart

# allow us to install RVM via curl
$ sudo apt-get install curl

# install rvm
$ curl -L get.rvm.io | bash -s stable

# install rvm dependencies
$ sudo apt-get install build-essential \
                       openssl \
                       libreadline6 \
                       libreadline6-dev \
                       curl \
                       git-core \
                       zlib1g \
                       zlib1g-dev \
                       libssl-dev \
                       libyaml-dev \
                       libsqlite3-dev \
                       sqlite3 \
                       libxml2-dev \
                       libxslt-dev \
                       autoconf \
                       libc6-dev \
                       ncurses-dev \
                       automake \
                       libtool \
                       bison \
                       subversion

# Install Ruby 1.9.3
$ rvm install 1.9.3

-----
# You should't have to do this bit but RVM was not a function so:
$ echo '[[ -s "$HOME/.rvm/scripts/rvm" ]] && . "$HOME/.rvm/scripts/rvm" # Load RVM function' >> ~/.bash_profile

# apply changes to current shell
$ source ~/.bash_profile
-----

# set 1.9.3 as default system Ruby
$ rvm --default use 1.9.3

# Install The Editor
$ sudo apt-get install vim 

# Install The Framework
$ gem install rails

# create new Rails application
$ rails new demo

# go to it
$ cd demo

# install application dependencies
$ bundle install

# boot up the server
$ rails s

# visit http://localhost:3000/
